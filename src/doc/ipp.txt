Using IPP
=========

	set IPPSAMPLEROOT=X:\some-directory\ipp-samples
	set IPPROOT=c:\Program Files (x86)\Intel\ComposerXE-2011\ipp
	set UICBINROOT=%UICROOT%\_bin\your-architecture\bin
	set UICLIBROOT=%UICROOT%\_bin\your-architecture\stublib
	
	cd %IPPSAMPLEROOT%\image-codecs\uic
	set LINKAGE=static
	build_ia32.bat cl9

	%UICROOT%\src\codec\image\jpeg2000\dec\include
